\category{\altype{SerializableType}}\\
\alexp{lift}: & \% $\to$ \altype{Integer} & Conversion to an integer\\
\end{exports}
\alpage{lift}
\Usage{\name~x}
\Signature{\%}{\altype{Integer}}
\Params{ {\em x} & \% & an element of the field\\ }
